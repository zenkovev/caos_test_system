Скрипт создаёт следующее поддерево основной файловой системы.
Для mergefs будут переданы директории A, B, C.
В скобках указан порядок, в котором создаются одноимённые файлы
с одинаковым путём.

.
├── A
│   ├── dir1
│   │   ├── tmp1.txt (2)
│   │   └── tmp2.txt
│   └── dir2
│       ├── a.txt
│       └── tmp1.txt
├── B
│   ├── dir1
│   │   ├── tmp1.txt (3)
│   │   └── tmp3.txt (1)
│   └── dir3
│       ├── c.txt
│       ├── d
│       └── e.txt
└── C
    ├── dir1
    │   ├── tmp1.txt (1)
    │   └── tmp3.txt (2)
    └── dir2
        ├── b.txt
        └── tmp2.txt
